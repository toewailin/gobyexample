package main

import "fmt"

ဒီ fact function က base case fact(0) ကို မရောက်မခြင်းသူ့ကိုယ်သူ ပြန်ခေါ်နေပါလိမ့်မယ်။

 
func fact(n int) int {
    if n == 0 {
        return 1
    }
    return n * fact(n-1)
}

    fmt.Println(fact(7))

Closures လဲ recursive ဖြစ်လို့ရတယ်၊ ဒါပေမယ့် closure function ကို defined မလုပ်ခင် type ကို var ဆိုပီးသီးသန့်ကြေငြာပေးရမယ်။

 
    var fib func(n int) int

    fib = func(n int) int {
        if n < 2 {
            return n
        }
