Standard library ရဲ့ strings package က အသုံးဝင်တဲ့ string-related function တွေ အများကြီး ပေးထားပါတယ်။ ဒီမှာ ဒီ package အကြောင်း ခန့်မှန်းသိနိုင်အောင် ဥပမာအချို့ ပြထားပါတယ်။

    p("Contains:  ", s.Contains("test", "es"))

    p("Split:     ", s.Split("a-b-c-d-e", "-"))
    p("ToLower:   ", s.ToLower("TEST"))
    p("ToUpper:   ", s.ToUpper("test"))
}
